ls           -показывеат все файлы и папки внутри текущей директории
ls test      -показывет все файлы и папки внутри папки test
ls  -a       -показывет также скрытые файлы и папки
ls  -l       -показывеат более подробную информацию

cd           -переходит в домашнюю папку
cd~          -переходитв домашнюю папки
cd test      -переходитв папку test
cd .         -переходит в текущуюю папку (остается в ней же)
cd ..        -переходит в предыдуюю папку
cd test/test2 -переходитв папку test2 которая находится в папке test
cd ../..      -переходит на 2 папки назад

mkdir test    -создает папку
mkdir test/test2 -создает папку тест2 внутри тест (если папка тест сушествует)

touch test.txt  -создает файл test.txt
touch test/test.txt -создает файл test.txt внутри папки test
touch test{1,2}.txt  -созадет файлы test1.txt и test2.txt

rm test             -удаляет файл test (если это не папка)
rm  -rf     test    -удаляет файл или папку со всем содержимым (удаляет безвозвратно)
rm  -rf /           !!! не писать никогда

pwd                 -показывает где мы находимся (print working direktory)

cat test.txt        -показывает содержимое файла
nano tstt.txt       -открывает tstt.txt в редакторе кода нано (в терминале)
code test.txt       -открывает test.txt в редакторе кода vscode

sudo                -права суперюзера (superuser do)
sudo apt install git  -установить систему git
(apt  -системный установшик пакетов)
sudo snap inststall code  --classic  -установить програмуу vscode
